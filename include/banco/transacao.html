SQLSERVER
DECLARE @erro int;
DECLARE @erro_msg int;
	SET @erro = 0;
	BEGIN TRAN excluiEsforco;
UPDATE tblEsforco SET 
	st_excluido='S'
WHERE 
	cd_esforco=".$cd_esforco."
SET @erro_msg = @@ERROR
IF @erro_msg <> 0 
SET @erro = @erro_msg;
IF @erro <> 0
	ROLLBACK TRAN excluiEsforco;
ELSE
	COMMIT TRAN excluiEsforco;
SELECT @erro AS st_transacao
